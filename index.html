// --- A NOVA MÁGICA DA IA (CORRIGIDA) --- // window.gerarRelatorioIA = async function(event) { if(event) event.preventDefault(); // 1. Captura os valores dos campos const prob = document.getElementById('os-problema').value; const serv = document.getElementById('os-servico').value; const cliente = window.clienteAtualOS || "Cliente Brasilguard"; if(!prob || !serv) return alert("Por favor, preencha o problema e a solução."); // 2. Feedback visual no botão const btn = event.target; const textoOriginal = btn.innerText; btn.innerText = "⏳ AGENTE HUNTER ESCREVENDO..."; btn.disabled = true; btn.style.opacity = "0.7"; try { // 3. Chamada para a Nave-Mãe (Render) const res = await fetch(`${BASE_URL}/api/hunter/gerar-os`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ cliente: cliente, problema: prob, servico: serv }) }); // 4. AGORA a variável 'dados' é criada com a resposta const dados = await res.json(); if(dados.sucesso) { // 5. Abre o WhatsApp com o relatório pronto const mensagem = encodeURIComponent(dados.relatorio); window.open(`https://wa.me/?text=${mensagem}`, '_blank'); window.fecharModal(); } else { alert("A IA teve um pequeno soluço: " + (dados.erro || "Verifique o console do Render.")); } } catch(e) { console.error("Erro na conexão:", e); alert("Erro na conexão com a Nave-Mãe. Aguarde o Render ligar ou verifique a internet."); } finally { // 6. Restaura o botão ao estado original btn.innerText = textoOriginal; btn.disabled = false; btn.style.opacity = "1"; } } // --- FUNÇÕES DO HUNTER (VENDAS) --- // window.salvarTreinamento = async function() { const texto = document.getElementById('texto-personalidade').value; try { const res = await fetch(`${BASE_URL}/api/hunter/treinar`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ instrucoes: texto, clienteId: window.usuarioLogadoId }) }); const dados = await res.json(); if(dados.sucesso) { const aviso = document.getElementById('aviso-salvo'); aviso.style.display = 'inline'; setTimeout(() => aviso.style.display = 'none', 3000); } } catch(e) { console.error("Erro ao salvar treinamento", e); } } window.carregarMemoria = async function() { try { const res = await fetch(`${BASE_URL}/api/hunter/memoria?clienteId=${window.usuarioLogadoId}`); const dados = await res.json(); if(dados.texto) document.getElementById('texto-personalidade').value = dados.texto; } catch(e) { console.error("Erro ao carregar memória", e); } } window.carregarLeads = async function() { try { const resposta = await fetch(`${BASE_URL}/api/leads?clienteId=${window.usuarioLogadoId}`); const leads = await resposta.json(); const tbody = document.getElementById('tabela-leads'); tbody.innerHTML = ''; if(leads.length === 0) { tbody.innerHTML = 'Cofre vazio. Capture leads no Maps.'; return; } leads.forEach(lead => { const tr = document.createElement('tr'); tr.innerHTML = `${lead.telefone}${lead.origem}${lead.status || 'Novo'}`; tbody.appendChild(tr); }); } catch (erro) { document.getElementById('tabela-leads').innerHTML = 'Erro na conexão com a Nave-Mãe.'; } }
